/* Program : MGRCABDTL                                              */
/* Purpose : Manager - View Cab Details (Callee)                    */
/********************************************************************/
ctl-opt dftactgrp(*no)
        actgrp(*caller);

/*------------------------------------------------------------------*/
/* Files                                                            */
/*------------------------------------------------------------------*/
dcl-f MGRCABDDSP workstn;
dcl-f CABLFMGR   usage(*input) keyed;

/*------------------------------------------------------------------*/
/* Parameter                                                        */
/*------------------------------------------------------------------*/
dcl-pi *n;
   pCabId packed(5:0);
end-pi;

/*------------------------------------------------------------------*/
/* Fetch Cab                                                        */
/*------------------------------------------------------------------*/
chain pCabId CABLFMGR;
if %notfound(CABLFMGR);
   return;
endif;

/*------------------------------------------------------------------*/
/* Load Display                                                     */
/*------------------------------------------------------------------*/
DSP_CABID  = CABID;
DSP_CABNO  = CABNO;
DSP_MODEL  = MODEL;
DSP_DRV    = DRIVERID;
DSP_STAT   = STATUS;

/*------------------------------------------------------------------*/
/* Show Window                                                      */
/*------------------------------------------------------------------*/
exfmt CABDTLW;

return;
