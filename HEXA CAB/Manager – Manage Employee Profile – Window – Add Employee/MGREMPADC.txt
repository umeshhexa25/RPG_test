ctl-opt dftactgrp(*no)
        actgrp(*caller);

/*------------------------------------------------------------------*/
/* Files                                                            */
/*------------------------------------------------------------------*/
dcl-f EMPLFMGRAD usage(*update) keyed;

/*------------------------------------------------------------------*/
/* Parameters (implicit from DSPF fields)                           */
/*------------------------------------------------------------------*/

/*------------------------------------------------------------------*/
/* Validate Duplicate Employee                                      */
/*------------------------------------------------------------------*/
chain DSP_EMP EMPLFMGRAD;

if %found(EMPLFMGRAD);
   return;
endif;

/*------------------------------------------------------------------*/
/* Populate Record                                                  */
/*------------------------------------------------------------------*/
EMPID   = DSP_EMP;
EMPNAME = DSP_NAME;
ADDRESS = DSP_ADDR;
PHONE   = DSP_PHONE;
STATUS  = DSP_STAT;

/*------------------------------------------------------------------*/
/* Write Employee                                                   */
/*------------------------------------------------------------------*/
write EMPREC;

return;
